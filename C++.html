🔰 基礎 C++ 觀念整理

1️⃣ 程式基本結構

#include <iostream>
using namespace std;

int main() {
    cout << "Hello, world!" << endl;
    return 0;
}

2️⃣ 變數與資料型別

型別意義範例
int整數int age = 20;
float浮點數float pi = 3.14;
double雙精度浮點double d = 3.1415;
char字元char c = 'A';
bool布林值bool isOpen = true;
string字串（需引入）string name = "Tom";
#include <iostream>
#include <string>
using namespace std;

int main() {
    int age = 18;
    string name = "Alice";
    cout << name << " is " << age << " years old." << endl;
    return 0;
}

3️⃣ 條件判斷

int score = 85;
if (score >= 60) {
    cout << "及格" << endl;
} else {
    cout << "不及格" << endl;
}

4️⃣ 迴圈

while 迴圈： 
int i = 0;
while (i < 5) {
    cout << i << " ";
    i++;
}
for 迴圈： 
for (int i = 0; i < 5; i++) {
    cout << i << " ";
}

5️⃣ 陣列（Array）

int nums[3] = {10, 20, 30};
cout << nums[1]; // 輸出 20

6️⃣ 函式（Function）

int add(int a, int b) {
    return a + b;
}

int main() {
    cout << add(3, 4); // 輸出 7
    return 0;
}

7️⃣ 類別（Class）與物件（Object）

class Student {
public:
    string name;
    int age;

    void sayHello() {
        cout << "Hi, I'm " << name << endl;
    }
};

int main() {
    Student s;
    s.name = "Amy";
    s.age = 19;
    s.sayHello();
    return 0;
}
